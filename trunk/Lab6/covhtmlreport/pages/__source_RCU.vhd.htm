    10  LIBRARY IEEE;
    11  USE IEEE.std_logic_1164.ALL;
    12  
    13  ENTITY rcu is
    14      port(
    15          CLK                    :        IN            std_logic;
    16          RST_N                :        IN            std_logic;
    17          D_EDGE            :        IN            std_logic;
    18          EOP                    :        IN            std_logic;
    19          SHIFT_ENABLE:        IN            std_logic;        
    20          RCV_DATA        :        IN            std_logic_vector(7 downto 0);        
    21          RCVING            :        OUT            std_logic;
    22          W_ENABLE        :        OUT            std_logic;        
    23          R_ERROR            :        OUT            std_logic);
    24          
    25  end rcu;
    26  
    27  architecture behav of rcu is
    28  type state_type is (idle, rcvwait, rcvsync, chksync, rcvbyte,fifostrobe,ierror,serror,eoperror, eopidle);
    29  signal state,nextstate : state_type;
    30  signal count : std_logic;
    31  signal cnt : integer range 0 to 8;
    32      begin
    33      
    34      statelogic: process(CLK, RST_N)
    35      begin
    36      
    37          if (RST_N = '0') then
    38              state <= idle;
    39              cnt<=0;
    40          elsif(rising_edge(clk)) then
    41              state <= nextstate;
    42              if ((SHIFT_ENABLE = '1') and (count = '1')) then
    43                  cnt<= cnt + 1;
    44              elsif(count = '0') then
    45                  cnt<=0;
    46              end if;
    47                  
    48          end if;
    49          
    50      end process statelogic;
    51      
    52      
    53      nextlogic: process(state, D_EDGE, EOP, SHIFT_ENABLE, RCV_DATA)
    54  
    55      begin
    56      
    57          nextstate <= state;
    58          
    59          case state is 
    60          
    61          when idle =>
    62          
    63              if (D_EDGE = '1') then
    64                  nextstate <= rcvwait;
    65              else
    66                  nextstate <= state;
    67                  
    68              end if;
    69  
    70              
    71          when rcvwait =>  -- check with your shift reg to see when it gets data ( either 1 cycle after or just at the moment);
    72          
    73              if (EOP = '1') then
    74                  nextstate <= ierror;
    75              else
    76                  nextstate <= rcvsync;
    77              end if;
    78              
    79          when rcvsync =>
    80          
    81              if (EOP = '1') then
    82                  nextstate <= ierror;
    83              elsif(cnt = 8) then
    84                  
    85                  nextstate <= chksync;
    86              end if;
    87              
    88          when chksync =>
    89          
    90              if (EOP = '1') then
    91                  nextstate <= ierror;
    92              elsif(RCV_DATA = "10000000") then
    93                  nextstate <= rcvbyte;
    94              else
    95                  nextstate <= serror;
    96              end if;
    97              
    98          when rcvbyte =>
    99          
   100              if (EOP = '1') then
   101                  nextstate <= ierror;
   102              elsif(cnt = 8) then
   103                  nextstate <= fifostrobe;
   104              end if;
   105              
   106          when fifostrobe =>
   107          
   108              if (EOP = '1') then
   109                  nextstate <= eopidle;
   110              else
   111                  nextstate <= rcvbyte;
   112              end if;
   113              
   114          when serror =>
   115          
   116              if (EOP = '1') then
   117                  nextstate <= ierror;
   118              else
   119                  nextstate <= state;
   120              end if;
   121              
   122              
   123          when ierror => 
   124              
   125              if (D_EDGE = '1') then
   126                  nextstate <= eoperror;
   127              else
   128                  nextstate <= state;
   129              end if;
   130          -- Added 2 New States, EOPIDLE, EOPERROR
   131          
   132          when eoperror => 
   133              if (D_EDGE = '1') then
   134                  nextstate <= rcvwait;
   135              else
   136                  nextstate <= state;
   137              end if;
   138          
   139          when eopidle =>
   140              if (D_EDGE = '1') then
   141                  nextstate <= idle;
   142              else
   143                  nextstate <= state;
   144              end if;
   145          
   146              
   147          end case;
   148          
   149      end process nextlogic;
   150      
   151  
   152      outlogic: process(state)
   153      begin
   154      
   155          case state is 
   156          
   157          when idle =>
   158              count <= '0';
   159              RCVING <= '0';
   160              W_ENABLE <= '0';
   161              R_ERROR <= '0';
   162              
   163          when rcvwait =>
   164              
   165              count <= '0';
   166              RCVING <= '1'; 
   167              W_ENABLE <= '0';
   168              R_ERROR <= '0';
   169              
   170          when rcvsync =>
   171              
   172              count <= '1';
   173              RCVING <= '1'; 
   174              W_ENABLE <= '0';
   175              R_ERROR <= '0';
   176  
   177          when chksync =>
   178              
   179              count <= '0';
   180              RCVING <= '1'; 
   181              W_ENABLE <= '0';
   182              R_ERROR <= '0';
   183  
   184          when rcvbyte =>
   185              count <= '1';
   186              RCVING <= '1'; 
   187              W_ENABLE <= '0';
   188              R_ERROR <= '0';
   189  
   190          when fifostrobe =>
   191              count <= '0';
   192              RCVING <= '1'; 
   193              W_ENABLE <= '1';
   194              R_ERROR <= '0';
   195  
   196          when ierror =>
   197          
   198              RCVING <= '0'; 
   199              W_ENABLE <= '0';
   200              R_ERROR <= '1';
   201              count <= '0';
   202  
   203          when serror =>
   204          
   205              RCVING <= '1'; 
   206              W_ENABLE <= '0';
   207              R_ERROR <= '1';
   208              count <= '0';
   209              -- added 2 new states, EOPERROR, EOPIDLE
   210          when eoperror =>
   211              RCVING <= '0';
   212              W_ENABLE <= '0';
   213              R_ERROR <='1';
   214              count <= '0';
   215          when eopidle =>
   216              RCVING <= '0';
   217              W_ENABLE <= '0';
   218              R_ERROR <='0';
   219              count <= '0';
   220  
   221              
   222          end case;
   223          
   224      end process outlogic;
   225      
   226  end behav;
